	CAM_return( 0 )
	CAM(20,99,198,33,177,0,0)

	//waitを入れないとオ`トセ`ブに失・工耄

	//再战用オ`トセ`ブ
	workG(GW_ENTRY_EVENT,SET,100)

	workG(GW_NOW_ROUTE,SET,31);//艾菲梅尔加

	EV_end()
}
