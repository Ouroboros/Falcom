//　　山道を登ろうとすると滑って落ちてしまう。
EV_STOP_01
{

	TK_begin()

//	if( F4_09_GoShrineMia )
	//{
	move(PLAYER,0,RUN2,2644,-1934,0,200)
	//move(PLAYER2,0,RUN2,2644,-1934,0,50)
	move(PLAYER2,0,RUN2,2632,-1874,0,50)
	wait_move(PLAYER)
	wait_move(PLAYER2)

	if( !F_RAGNA )
	{
		if(GW_MAN==1)
		{
		//　　※ラグナ先頭の場合
			MES(RAGNA,"っと……",0)
			MES(RAGNA,"こんな場所で\n遊んでる場合じゃねえな。",0)
			MES(RAGNA,"あの城に急がねえと。",0)
			MES_close(RAGNA)
		}
		else
		{
		//　　※アルウェン先頭の場合
			MES(PARTNER,"むむ……滑るのう……",0)
			MES(PARTNER,"このような場所で\n道草を食っておる場合ではない。",0)
			MES(PARTNER,"わがムーンブリア城へ急がねばな。",0)
			MES_close(PARTNER)
		}
		F_set(RAGNA)
	}
	//}

	TK_end()
}
