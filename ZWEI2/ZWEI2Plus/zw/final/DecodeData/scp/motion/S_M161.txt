	//mF0 = 用カウンタ（壁接触判定）
	//mF1 = 用カウンタ（步行rgy）
	//mF2 = 壁接触判定
	//mF3 = 自爆k婴筏浚


	//★自爆してもいい？
	if_EV DAMAGE
	{
		if_HP 10 0
		{
			if_mF3= 0
			{
				mF3_set -1 -1 1

				E_charbase0			-1 -1 2108 100 0 0		//＠振樱ù螅２）
				E_charbase0			-1 -1 2145 100 0 0		//＠フラッシュ

				H_@csp -1 -1 10 0 0		//以降、ノ`ダメ`ジ

				//     set deg tim rnd len next
				B_NEXT  -1  -1   0   0   0   99
			}
		}
	}


	//★ダメ`ジ食らうたびに何か出す
	if_EV DAM_KEN
	{
		if_RAND 2
			E_csp3		-1 -1 1340 100 0 0		//＠粉
		el_RAND 2
			E_csp4		-1 -1 1340 100 0 0		//＠粉
		else
			E_Bone_4	-1 -1 1340 100 0 0		//＠粉
	}


	//★ホバ`演出
	if_SET 00 01 02
	{
		if_time 5
		{
			E_charbase0		-1 -1  570 60 0 0 		//＠土
		}
	}


	//------------------------------------------------------------------
	// 登场
	//------------------------------------------------------------------
	//     set deg tim rnd  len next
	B_NEXT  80   0  60   0    0   02	//□始


	//------------------------------------------------------------------
	// 待机
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  02   0   0   8  00  00
		//     set deg tim rnd len next
		B_NEXT  02   0  30   0   0   01		//□步行


	//------------------------------------------------------------------
	// 步行
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  01   2   4   8  60  00	//右周り
	B_MOV  00   2   4   8 -60  00	//左周り

	if_SET 01 00
	{
		if_TIME 1 0
		{
			mF2_set -1 -1 0
			
			mF1_add -1 -1 1		//膜盲炕厥＋＋
		}

		if_mF1= 7
		{
			mF1_set -1 -1 0		//★用カウンタ初期化

			if_HP 100 0
			{
				//     set deg tim rnd  len next
				B_NEXT  -1   0   0   0    0   03	//□放射
			}
		}

		if_EV TOUCH_MAP
		{
			if_mF2= 0
			{
				mF2_set -1 -1 1
				
				if_mF0= 1
				{
					mF0_set -1 -1 0
				}
				else
				{
					mF0_set -1 -1 1
				}
			}
		}
		
		if_mF0= 1
		{
			//     set deg tim rnd  len next
			B_NEXT  -1   0  32   0    0   01	//□右周り
		}
		else
		{
			//     set deg tim rnd  len next
			B_NEXT  -1   0  32   0    0   00	//□左周り
		}
	}

	//------------------------------------------------------------------
	// 攻模ǚ派）
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  03   2   0   8  60  00
		//     set deg tim rnd len next
		B_NEXT  03   0  32   0   0   01		//□步行


	//------------------------------------------------------------------
	// 攻模ㄗ员）
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2

		}

		}
