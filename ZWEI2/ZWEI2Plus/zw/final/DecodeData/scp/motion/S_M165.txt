////////////////////////////////////////////////////////////////////////
// モ`ションセット定x

	//------------------------------------------------------------------
	// 登场
	//------------------------------------------------------------------
	SET 80 80 0001 0020 -1 00	//登场

	//------------------------------------------------------------------
	// 待机
	//------------------------------------------------------------------
	SET 02 02 0001 0020 -1 00

	//------------------------------------------------------------------
	// 步行
	//------------------------------------------------------------------
	SET 01 02 0001 0020 -1 00

	//------------------------------------------------------------------
	// 攻模ㄍ模
	//------------------------------------------------------------------
	SET 10 10 0001 0020 -1 00
	SET 11 11 0001 0020 -1 00
		E_csp1			11 0001 31010 20 0 20 0		//＠アタリ



////////////////////////////////////////////////////////////////////////
// ル`チン分岐

	//------------------------------------------------------------------
	// 登场
	//------------------------------------------------------------------
	//     set deg tim rnd  len next
	B_NEXT  80   0   5   0    0   02	//□始

	//------------------------------------------------------------------
	// 待机
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  02   0   0   8  00  00
		//     set deg tim rnd len next
		B_NEXT  02   0   1  80   0   10		//□突模àまぐれにクイックタ`ン）
		B_NEXT  02   0  45   0   0   10		//□突

	//------------------------------------------------------------------
	// 突
	//------------------------------------------------------------------
	//    set typ spd rot xx1 xx2
	B_MOV  10   0   0  24  00  00
		//     set deg tim rnd len next
		B_NEXT  10   0  30   0   0   11		//□突_後半

	//    set typ spd rot xx1 xx2
	B_MOV  11   0  50   0  00  00
		//     set deg tim rnd len next
		B_NEXT  11   0  30   0   0   02		//□待机

	if_SET 11
	{
		if_TIME 0 0
		{
			OTO -1 -1 441 0 0 0		//突模樱
		}

		if_EV TOUCH_MAP
		{
			//     set deg tim rnd len next
			B_NEXT  -1   0   0   0   0   02		//□待机
		}

//		if_EV TOUCH_CH
		if_EV HIT
		{
			E_ADL -1 -1 1180 100 -1 -1		//＠ぶちかましn
			
			//     set deg tim rnd len next
			B_NEXT  -1   0   0   0   0   02		//□待机
		}
	}

	if_SET 10 11
	{
		E_charbase0		-1 0006  1245 100 0 0		//＠土
	}

	if_SET 11
	{
		if_HP 30 0
		{
			//道中に爆を置いていく
			if_TIME 15 15
			{
				G_charbase0 	-1 -1 3 56		//3,56:爆落とす
			}
		}
	}
